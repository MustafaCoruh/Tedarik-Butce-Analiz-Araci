'=========================================================
' TEDARİK KALEMLERİNİ PROJE / STATÜ / ÖZEL DOSYALARA AYIRIR
'
' Dinamik yapı:
' - Kaynak dosyada ilk 2 satır yok sayılır.
' - 3. satır başlık kabul edilir.
' - "Talep Eden Bölüm" sütunundaki birimler bulunur.
' - Kullanıcı frmBirimSecimi üzerinden bir veya birden fazla birim seçer.
' - Seçilen her birim için ayrı çıktı dosyaları oluşturulur.
'=========================================================
Public Sub VerileriIsle()

    Dim secilenDosya As String

    Dim kaynakWB As Workbook
    Dim kaynakWS As Worksheet

    Dim veriArr As Variant
    Dim birimArr As Variant

    Dim dictBirimler As Object
    Dim secilenBirimler As Object
    Dim frm As frmBirimSEcimi

    Dim colBirim As Long
    Dim colTarih As Long

    Dim masaustuYolu As String
    Dim tarihDamgasi As String
    Dim anaKlasor As String
    Dim birimKlasoru As String
    

    Dim birim As Variant
    Dim olusanDosyaSayisi As Long
    Dim islenenBirimSayisi As Long

    On Error GoTo HataYakala

secilenDosya = DosyaSec_Excel( _
                "Lütfen tasniflenecek Tedarik Kalemleri Excel dosyasını seçiniz.", _
                "Tedarik Kalemleri Dosyasını Seçiniz", _
                "Tedarik Kalemleri Dosyasını Seçiniz")

If Len(secilenDosya) = 0 Then Exit Sub
If gTopluCalismaAktif Then
   gSonTedarikKalemleriDosyaYolu = secilenDosya
End If



    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    Application.EnableEvents = False
    Application.Calculation = xlCalculationManual

    ' Kaynak dosya sadece okunur açılır; ilk 2 satır fiziksel olarak silinmez.
    Set kaynakWB = Workbooks.Open(secilenDosya, ReadOnly:=True)
    Set kaynakWS = kaynakWB.Sheets(1)

    ' 3. satır başlık kabul edilerek veri diziye alınır.
    veriArr = KaynakVeriyiDiziyeAl(kaynakWS, 3)

    kaynakWB.Close SaveChanges:=False

    colBirim = ColIxByHeaderAny(veriArr, Array("Talep Eden Bölüm"), 0)

    If colBirim = 0 Then
        MsgBox "'Talep Eden Bölüm' sütunu bulunamadı.", vbCritical, "Eksik Sütun"
        GoTo SafeExit
    End If

    colTarih = ColIxByHeaderAny(veriArr, Array("Huma Son Gözden Geçirme Tarihi", "HUMA Son Gözden Geçirme Tarihi"), 40)

    Set dictBirimler = GetUniqueBirimlerFromArray(veriArr, colBirim)

    If dictBirimler Is Nothing Or dictBirimler.Count = 0 Then
        MsgBox "'Talep Eden Bölüm' sütununda birim bilgisi bulunamadı.", vbCritical, "Veri Bulunamadı"
        GoTo SafeExit
    End If

    ' Kullanıcıya birim seçtir.
    Application.ScreenUpdating = True

    Set frm = New frmBirimSEcimi
    frm.BirimleriYukle dictBirimler
    frm.Show

    If frm.IptalEdildi Then
        Unload frm
        GoTo SafeExit
    End If

    Set secilenBirimler = frm.secilenBirimler
    Unload frm

    If secilenBirimler Is Nothing Or secilenBirimler.Count = 0 Then
        MsgBox "Seçili birim bulunamadı. İşlem iptal edildi.", vbExclamation, "Seçim Yok"
        GoTo SafeExit
    End If

    Application.ScreenUpdating = False

    masaustuYolu = Cikti_KaynakKlasoruAl(secilenDosya)
    tarihDamgasi = Format(Date, "dd.mm.yyyy")
    
    anaKlasor = Cikti_KlasoruAl("01 - Tedarik Kalemlerini Tasnifleme", masaustuYolu)




    For Each birim In secilenBirimler.Keys

        birimArr = FilterArrayByColumnValue(veriArr, colBirim, CStr(birim))

        If IsArray(birimArr) Then

            If UBound(birimArr, 1) >= 2 Then

                If colTarih > 0 Then
                    SortArrayByDate birimArr, colTarih
                End If

    birimKlasoru = anaKlasor & "\" & ShortFilePart(CStr(birim), 45)

    Cikti_EnsureFolderExists birimKlasoru


Cikti_EnsureFolderExists birimKlasoru


olusanDosyaSayisi = olusanDosyaSayisi + _
                     BirimDosyalariniOlustur( _
                        birimArr, _
                        CStr(birim), _
                        birimKlasoru, _
                        tarihDamgasi)


                islenenBirimSayisi = islenenBirimSayisi + 1

            End If

        End If

    Next birim

    MsgBox "Tedarik kalemleri tasnifleme işlemi başarıyla tamamlanmıştır." & vbCrLf & vbCrLf & _
           "Seçilen birim sayısı: " & secilenBirimler.Count & vbCrLf & _
           "İşlenen birim sayısı: " & islenenBirimSayisi & vbCrLf & _
           "Oluşturulan dosya sayısı: " & olusanDosyaSayisi & vbCrLf & vbCrLf & _
           "Dosyalar aşağıdaki klasöre kaydedilmiştir:" & vbCrLf & _
anaKlasor, _
           vbInformation, "İşlem Tamamlandı"

SafeExit:

    Application.ScreenUpdating = True
    Application.DisplayAlerts = True
    Application.EnableEvents = True
    Application.Calculation = xlCalculationAutomatic

    Exit Sub

HataYakala:

    On Error Resume Next

    If Not kaynakWB Is Nothing Then
        kaynakWB.Close SaveChanges:=False
    End If

    Application.ScreenUpdating = True
    Application.DisplayAlerts = True
    Application.EnableEvents = True
    Application.Calculation = xlCalculationAutomatic

    On Error GoTo 0

    MsgBox "Tedarik ayırma makrosunda hata oluştu:" & vbCrLf & vbCrLf & _
           Err.Description, _
           vbCritical, "Hata"

End Sub

'=========================================================
' 3. SATIRI BAŞLIK KABUL EDEREK KAYNAK VERİYİ DİZİYE ALIR
'=========================================================
Private Function KaynakVeriyiDiziyeAl(ByVal ws As Worksheet, _
                                      ByVal headerRow As Long) As Variant

    Dim lastRow As Long
    Dim lastCol As Long

    lastRow = ws.Cells.Find(What:="*", _
                            After:=ws.Range("A1"), _
                            LookAt:=xlPart, _
                            LookIn:=xlFormulas, _
                            SearchOrder:=xlByRows, _
                            SearchDirection:=xlPrevious, _
                            MatchCase:=False).Row

    lastCol = ws.Cells(headerRow, ws.Columns.Count).End(xlToLeft).Column

    If lastRow < headerRow Then
        Err.Raise vbObjectError + 1001, , "Kaynak dosyada okunacak veri bulunamadı."
    End If

    KaynakVeriyiDiziyeAl = ws.Range(ws.Cells(headerRow, 1), ws.Cells(lastRow, lastCol)).Value

End Function

'=========================================================
' DİZİDEN BENZERSİZ BİRİMLERİ BULUR
'=========================================================
Private Function GetUniqueBirimlerFromArray(ByVal veriArr As Variant, _
                                            ByVal colBirim As Long) As Object

    Dim dict As Object
    Dim i As Long
    Dim birimAdi As String

    Set dict = CreateObject("Scripting.Dictionary")
    dict.CompareMode = vbTextCompare

    For i = 2 To UBound(veriArr, 1)

        birimAdi = Trim$(CStr(veriArr(i, colBirim)))

        If Len(birimAdi) > 0 Then

            If Not dict.Exists(birimAdi) Then
                dict.Add birimAdi, True
            End If

        End If

    Next i

    Set GetUniqueBirimlerFromArray = dict

End Function

'=========================================================
' SEÇİLEN BİRİME GÖRE DİZİYİ FİLTRELER
'=========================================================
Private Function FilterArrayByColumnValue(ByVal veriArr As Variant, _
                                          ByVal colIndex As Long, _
                                          ByVal filterValue As String) As Variant

    Dim i As Long
    Dim j As Long
    Dim r As Long

    Dim rowCount As Long
    Dim colCount As Long
    Dim resultArr() As Variant

    colCount = UBound(veriArr, 2)

    For i = 2 To UBound(veriArr, 1)

        If StrComp(Trim$(CStr(veriArr(i, colIndex))), Trim$(filterValue), vbTextCompare) = 0 Then
            rowCount = rowCount + 1
        End If

    Next i

    If rowCount = 0 Then Exit Function

    ReDim resultArr(1 To rowCount + 1, 1 To colCount)

    ' Başlık satırı
    For j = 1 To colCount
        resultArr(1, j) = veriArr(1, j)
    Next j

    r = 1

    For i = 2 To UBound(veriArr, 1)

        If StrComp(Trim$(CStr(veriArr(i, colIndex))), Trim$(filterValue), vbTextCompare) = 0 Then

            r = r + 1

            For j = 1 To colCount
                resultArr(r, j) = veriArr(i, j)
            Next j

        End If

    Next i

    FilterArrayByColumnValue = resultArr

End Function

'=========================================================
' SEÇİLEN BİRİM İÇİN TÜM DOSYALARI OLUŞTURUR
'=========================================================
Private Function BirimDosyalariniOlustur(ByVal veriArr As Variant, _
                                         ByVal birimAdi As String, _
                                         ByVal kayitKlasoru As String, _
                                         ByVal tarihDamgasi As String) As Long

    Dim projeWB As Workbook
    Dim statuWB As Workbook

    Dim dosyaSayisi As Long
    Dim birimDosyaAdi As String

    On Error GoTo ErrHandler

    birimDosyaAdi = CleanFilePart(birimAdi)

    ' Projeye göre ayır
    Set projeWB = Workbooks.Add(xlWBATWorksheet)

    ProjeyeGoreAyir projeWB, veriArr
    Sayfa1iGuvenliSil projeWB

    projeWB.SaveAs Filename:=MakeUniquePath(kayitKlasoru & "\" & _
                                            birimDosyaAdi & " Tedarik Takibi Proje " & _
                                            tarihDamgasi & ".xlsx"), _
                    FileFormat:=xlOpenXMLWorkbook

    projeWB.Close SaveChanges:=False
    dosyaSayisi = dosyaSayisi + 1

    ' Statüye göre ayır
    Set statuWB = Workbooks.Add(xlWBATWorksheet)

    StatuyeGoreAyir statuWB, veriArr
    Sayfa1iGuvenliSil statuWB

    dosyaSayisi = dosyaSayisi + OzelDosyalariOlustur( _
                                    statuWB, _
                                    kayitKlasoru, _
                                    tarihDamgasi, _
                                    birimAdi)

    statuWB.SaveAs Filename:=MakeUniquePath(kayitKlasoru & "\" & _
                                            birimDosyaAdi & " Tedarik Takibi Statu " & _
                                            tarihDamgasi & ".xlsx"), _
                    FileFormat:=xlOpenXMLWorkbook

    statuWB.Close SaveChanges:=False
    dosyaSayisi = dosyaSayisi + 1

    BirimDosyalariniOlustur = dosyaSayisi
    Exit Function

ErrHandler:

   Dim hataMesaji As String

   hataMesaji = Err.Description

   On Error Resume Next

   If Not projeWB Is Nothing Then projeWB.Close SaveChanges:=False
   If Not statuWB Is Nothing Then statuWB.Close SaveChanges:=False

   On Error GoTo 0

   MsgBox "Aşağıdaki birim için dosya oluşturulamadı:" & vbCrLf & vbCrLf & _
          birimAdi & vbCrLf & vbCrLf & _
          "Hata açıklaması:" & vbCrLf & _
          hataMesaji, _
          vbCritical, "Tedarik Tasnifleme Hatası"

   BirimDosyalariniOlustur = 0


End Function

'=========================================================
' PROJEYE GÖRE AYIRMA
'=========================================================
Private Sub ProjeyeGoreAyir(ByVal wb As Workbook, _
                            ByVal veriArr As Variant)

    Dim i As Long
    Dim satir As Long
    Dim sutunSayisi As Long

    Dim projeAdi As String
    Dim ws As Worksheet
    Dim projeler As Object
    Dim colProje As Long

    Set projeler = CreateObject("Scripting.Dictionary")
    projeler.CompareMode = vbTextCompare

    sutunSayisi = UBound(veriArr, 2)

    colProje = ColIxByHeaderAny(veriArr, Array("Proje"), 11)

    For i = 2 To UBound(veriArr, 1)

        projeAdi = CStr(veriArr(i, colProje))

        If Len(Trim$(projeAdi)) = 0 Then
            projeAdi = "(Boş Proje)"
        End If

        If Not projeler.Exists(projeAdi) Then

            Set ws = wb.Sheets.Add(After:=wb.Sheets(wb.Sheets.Count))

            On Error Resume Next
            ws.Name = Left$(TemizSayfaAdi(projeAdi), 31)

            If Err.Number <> 0 Then
                Err.Clear
                ws.Name = "Proje_" & i
            End If

            On Error GoTo 0

            ws.Range("A1").Resize(1, sutunSayisi).Value = Application.Index(veriArr, 1, 0)

            projeler.Add projeAdi, ws

        End If

        Set ws = projeler(projeAdi)

        satir = ws.Cells(ws.rows.Count, 1).End(xlUp).Row + 1
        ws.Range("A" & satir).Resize(1, sutunSayisi).Value = Application.Index(veriArr, i, 0)

    Next i

    For Each ws In wb.Sheets
        ZSutununuMetneCevir ws
        TabloyaCevir ws
    Next ws

End Sub

'=========================================================
' BAŞLIK / SATIR YAZMA
'=========================================================
Private Sub BaslikYaz(ByVal ws As Worksheet, _
                      ByVal veriArr As Variant, _
                      ByVal kolonlar As Variant)

    Dim i As Long
    Dim hedefCol As Long

    For i = LBound(kolonlar) To UBound(kolonlar)

        hedefCol = i - LBound(kolonlar) + 1

        If CLng(kolonlar(i)) > 0 Then
            ws.Cells(1, hedefCol).Value = veriArr(1, CLng(kolonlar(i)))
        Else
            ws.Cells(1, hedefCol).Value = "(Başlık Bulunamadı)"
        End If

    Next i

End Sub

Private Sub SatirYaz(ByVal ws As Worksheet, _
                     ByVal veriArr As Variant, _
                     ByVal satirIndex As Long, _
                     ByVal kolonlar As Variant)

    Dim i As Long
    Dim hedefSatir As Long
    Dim hedefCol As Long

    hedefSatir = ws.Cells(ws.rows.Count, 1).End(xlUp).Row + 1

    For i = LBound(kolonlar) To UBound(kolonlar)

        hedefCol = i - LBound(kolonlar) + 1

        If CLng(kolonlar(i)) > 0 Then
            ws.Cells(hedefSatir, hedefCol).Value = veriArr(satirIndex, CLng(kolonlar(i)))
        Else
            ws.Cells(hedefSatir, hedefCol).Value = ""
        End If

    Next i

End Sub

'=========================================================
' ÖZEL DOSYALAR
'=========================================================
Private Function OzelDosyalariOlustur(ByVal statuWB As Workbook, _
                                      ByVal kayitKlasoru As String, _
                                      ByVal tarihDamgasi As String, _
                                      ByVal birimAdi As String) As Long

    Dim adet As Long

    If TeslimatiGecikmisDosyasiOlustur(statuWB, kayitKlasoru, tarihDamgasi, birimAdi) Then
        adet = adet + 1
    End If

    If Huma1HaftaBekleyenDosyasiOlustur(statuWB, kayitKlasoru, tarihDamgasi, birimAdi) Then
        adet = adet + 1
    End If

    If Reddedilen2IsGunuDosyasiOlustur(statuWB, kayitKlasoru, tarihDamgasi, birimAdi) Then
        adet = adet + 1
    End If

    OzelDosyalariOlustur = adet

End Function

'=========================================================
' 1) TESLİMATI GECİKMİŞLER
'=========================================================
Private Function TeslimatiGecikmisDosyasiOlustur(ByVal statuWB As Workbook, _
                                                 ByVal kayitKlasoru As String, _
                                                 ByVal tarihDamgasi As String, _
                                                 ByVal birimAdi As String) As Boolean

    Dim kaynakWS As Worksheet
    Dim yeniWB As Workbook
    Dim yeniWS As Worksheet

    Dim sonSatir As Long
    Dim sonSutun As Long
    Dim i As Long
    Dim hedefSatir As Long
    Dim statuText As String
    Dim birimDosyaAdi As String

    On Error GoTo ErrHandler

    birimDosyaAdi = CleanFilePart(birimAdi)

    Set kaynakWS = statuWB.Sheets("Teslim Edilecek Talepler")

    sonSatir = kaynakWS.Cells(kaynakWS.rows.Count, "A").End(xlUp).Row
    sonSutun = kaynakWS.Cells(1, kaynakWS.Columns.Count).End(xlToLeft).Column

    If sonSatir < 2 Then Exit Function

    Set yeniWB = Workbooks.Add(xlWBATWorksheet)
    Set yeniWS = yeniWB.Sheets(1)

    kaynakWS.Range(kaynakWS.Cells(1, 1), kaynakWS.Cells(1, sonSutun)).Copy yeniWS.Range("A1")

    hedefSatir = 2

    For i = 2 To sonSatir

        statuText = NormalizeString(CStr(kaynakWS.Cells(i, 3).Value))

        If statuText = "kismi teslimat" Or statuText = "teslimat gecikti" Then

            kaynakWS.Range(kaynakWS.Cells(i, 1), kaynakWS.Cells(i, sonSutun)).Copy yeniWS.Cells(hedefSatir, 1)
            hedefSatir = hedefSatir + 1

        End If

    Next i

    If hedefSatir = 2 Then
        yeniWB.Close SaveChanges:=False
        Exit Function
    End If

    TabloyaCevir yeniWS

    yeniWB.SaveAs Filename:=MakeUniquePath(kayitKlasoru & "\" & _
                                           birimDosyaAdi & " Teslimatı Gecikmiş Tedarik Kalemler " & _
                                           tarihDamgasi & ".xlsx"), _
                   FileFormat:=xlOpenXMLWorkbook

    yeniWB.Close SaveChanges:=False

    TeslimatiGecikmisDosyasiOlustur = True
    Exit Function

ErrHandler:

    On Error Resume Next
    If Not yeniWB Is Nothing Then yeniWB.Close SaveChanges:=False
    On Error GoTo 0

    TeslimatiGecikmisDosyasiOlustur = False

End Function

'=========================================================
' 2) 1 HAFTA ÖNCESİ HUMA SÜRECİ
'=========================================================
Private Function Huma1HaftaBekleyenDosyasiOlustur(ByVal statuWB As Workbook, _
                                                  ByVal kayitKlasoru As String, _
                                                  ByVal tarihDamgasi As String, _
                                                  ByVal birimAdi As String) As Boolean

    Dim kaynakWS As Worksheet
    Dim yeniWB As Workbook
    Dim yeniWS As Worksheet

    Dim sonSatir As Long
    Dim sonSutun As Long
    Dim i As Long
    Dim hedefSatir As Long

    Dim kontrolTarihi As Date
    Dim hucreTarih As Variant
    Dim eskiTarih As String
    Dim birimDosyaAdi As String

    On Error GoTo ErrHandler

    birimDosyaAdi = CleanFilePart(birimAdi)

    Set kaynakWS = statuWB.Sheets("HUMA Süreci Devam Eden Talepler")

    kontrolTarihi = Date - 7
    eskiTarih = Format(kontrolTarihi, "dd.mm.yyyy")

    sonSatir = kaynakWS.Cells(kaynakWS.rows.Count, "A").End(xlUp).Row
    sonSutun = kaynakWS.Cells(1, kaynakWS.Columns.Count).End(xlToLeft).Column

    If sonSatir < 2 Then Exit Function

    With kaynakWS.Sort
        .SortFields.Clear
        .SortFields.Add key:=kaynakWS.Range("K1"), _
                        SortOn:=xlSortOnValues, _
                        Order:=xlAscending, _
                        DataOption:=xlSortNormal

        .SetRange kaynakWS.Range("A1").CurrentRegion
        .Header = xlYes
        .Apply
    End With

' DİZİYİ TARİHE GÖRE SIRALAR
'=========================================================
Private Sub SortArrayByDate(ByRef arr As Variant, _
                            ByVal dateCol As Long)

    Dim i As Long
    Dim j As Long

    If dateCol <= 0 Then Exit Sub
    If UBound(arr, 1) < 3 Then Exit Sub

    For i = 2 To UBound(arr, 1) - 1

        For j = i + 1 To UBound(arr, 1)

            If DateSortKey(arr(i, dateCol)) > DateSortKey(arr(j, dateCol)) Then
                SwapArrayRows arr, i, j
            End If

        Next j

    Next i

End Sub

Private Function DateSortKey(ByVal v As Variant) As Double

    Dim d As Date

    If TryGetDate(v, d) Then
        DateSortKey = CDbl(d)
    Else
        DateSortKey = 99999999
    End If

End Function

Private Sub SwapArrayRows(ByRef arr As Variant, _
                          ByVal rowA As Long, _
                          ByVal rowB As Long)

    Dim c As Long
    Dim temp As Variant

    For c = 1 To UBound(arr, 2)

        temp = arr(rowA, c)
        arr(rowA, c) = arr(rowB, c)
        arr(rowB, c) = temp

    Next c

End Sub

Private Sub TabloyaCevir(ByVal ws As Worksheet)

    On Error Resume Next

    Dim veriAralik As Range
    Dim sonSatir As Long
    Dim sonSutun As Long
    Dim lo As ListObject

    For Each lo In ws.ListObjects
        lo.Delete
    Next lo

    sonSatir = ws.Cells(ws.rows.Count, "A").End(xlUp).Row
    sonSutun = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column

    If sonSatir < 2 Then Exit Sub

    Set veriAralik = ws.Range(ws.Cells(1, 1), ws.Cells(sonSatir, sonSutun))

    Set lo = ws.ListObjects.Add(xlSrcRange, veriAralik, , xlYes)

    lo.Name = SafeTableName("tbl_" & ws.Index & "_" & ws.Name)

    veriAralik.Columns.AutoFit

    On Error GoTo 0

End Sub

                       Semicolon:=False, _
                       Comma:=False, _
                       Space:=False, _
                       Other:=False, _
                       FieldInfo:=Array(1, xlTextFormat), _
                       TrailingMinusNumbers:=True

    End With

End Sub

Private Function ColIxByHeaderAny(ByVal veriArr As Variant, _
                                  ByVal headerNames As Variant, _
                                  Optional ByVal fallback As Long = 0) As Long

    Dim h As Variant
    Dim ix As Long

    For Each h In headerNames

        ix = ColIxByHeader(veriArr, CStr(h), 0)

        If ix > 0 Then
            ColIxByHeaderAny = ix
            Exit Function
        End If

    Next h

    ColIxByHeaderAny = fallback

End Function

Private Function ColIxByHeader(ByVal veriArr As Variant, _
                               ByVal headerName As String, _
                               Optional ByVal fallback As Long = 0) As Long

    Dim j As Long

    For j = 1 To UBound(veriArr, 2)

        If NormalizeString(CStr(veriArr(1, j))) = NormalizeString(headerName) Then
            ColIxByHeader = j
            Exit Function
        End If

    Next j

    ColIxByHeader = fallback

End Function

Private Function NormalizeString(ByVal s As String) As String

    Dim t As String

    t = LCase$(Trim$(CStr(s)))
    t = Replace(t, Chr(160), " ")

    t = Replace(t, "ı", "i")
    t = Replace(t, "İ", "i")
    t = Replace(t, "ş", "s")
    t = Replace(t, "Ş", "s")
    t = Replace(t, "ğ", "g")
    t = Replace(t, "Ğ", "g")
    t = Replace(t, "ç", "c")
    t = Replace(t, "Ç", "c")
    t = Replace(t, "ö", "o")
    t = Replace(t, "Ö", "o")
    t = Replace(t, "ü", "u")
    t = Replace(t, "Ü", "u")

    Do While InStr(t, "  ") > 0
        t = Replace(t, "  ", " ")
    Loop

    NormalizeString = t

End Function

Private Function TryGetDate(ByVal v As Variant, _
                            ByRef outDate As Date) As Boolean

    Dim s As String
    Dim p As Variant

    On Error GoTo Failed

    If IsDate(v) Then
        outDate = CDate(v)
        TryGetDate = True
        Exit Function
    End If

    If IsNumeric(v) Then

        If CDbl(v) > 20000 And CDbl(v) < 80000 Then
            outDate = DateSerial(1899, 12, 30) + CDbl(v)
            TryGetDate = True
            Exit Function
        End If

    End If

    s = Trim$(CStr(v))

    If Len(s) = 0 Then GoTo Failed

    If InStr(1, s, " ", vbTextCompare) > 0 Then
        s = Split(s, " ")(0)
    End If

    s = Replace(s, "/", ".")
    s = Replace(s, "-", ".")

    p = Split(s, ".")

    If UBound(p) = 2 Then

        If Len(CStr(p(2))) = 4 Then

            outDate = DateSerial(CLng(p(2)), CLng(p(1)), CLng(p(0)))
            TryGetDate = True
            Exit Function

        End If

    End If

Failed:

    TryGetDate = False

End Function

' DOSYA / SAYFA / TABLO ADI TEMİZLEME
'=========================================================
Private Function TemizSayfaAdi(ByVal s As String) As String

    Dim badChars As Variant
    Dim i As Long

    badChars = Array("/", "\", "[", "]", ":", "*", "?", """")

    s = Trim$(CStr(s))

    For i = LBound(badChars) To UBound(badChars)
        s = Replace$(s, badChars(i), " ")
    Next i

    If Len(s) = 0 Then s = "Sayfa"

    TemizSayfaAdi = s

End Function

Private Function CleanFilePart(ByVal filePart As String) As String

    Dim badChars As Variant
    Dim i As Long

    badChars = Array("\", "/", ":", "*", "?", """", "<", ">", "|")

    filePart = Trim$(CStr(filePart))

    For i = LBound(badChars) To UBound(badChars)
        filePart = Replace$(filePart, badChars(i), "-")
    Next i

    If Len(filePart) = 0 Then filePart = "Birim"

    CleanFilePart = filePart

End Function

Private Function SafeTableName(ByVal nm As String) As String

    nm = Trim$(CStr(nm))

    nm = Replace(nm, " ", "_")
    nm = Replace(nm, "-", "_")
    nm = Replace(nm, ".", "_")
    nm = Replace(nm, "/", "_")
    nm = Replace(nm, "\", "_")
    nm = Replace(nm, ":", "_")
    nm = Replace(nm, "?", "_")
    nm = Replace(nm, "*", "_")
    nm = Replace(nm, "[", "_")
    nm = Replace(nm, "]", "_")
    nm = Replace(nm, "(", "_")
    nm = Replace(nm, ")", "_")

    nm = Replace(nm, "ı", "i")
    nm = Replace(nm, "İ", "I")
    nm = Replace(nm, "ğ", "g")
    nm = Replace(nm, "Ğ", "G")
    nm = Replace(nm, "ü", "u")
    nm = Replace(nm, "Ü", "U")
    nm = Replace(nm, "ş", "s")
    nm = Replace(nm, "Ş", "S")
    nm = Replace(nm, "ö", "o")
    nm = Replace(nm, "Ö", "O")
    nm = Replace(nm, "ç", "c")
    nm = Replace(nm, "Ç", "C")

    If Len(nm) = 0 Then nm = "tbl"

    If Len(nm) > 200 Then
        nm = Left$(nm, 200)
    End If

    SafeTableName = nm

End Function

Private Function MakeUniquePath(ByVal fullPath As String) As String

    Dim base As String
    Dim ext As String
    Dim candidate As String
    Dim p As Long
    Dim i As Long

    p = InStrRev(fullPath, ".")

    If p > 0 Then
        base = Left$(fullPath, p - 1)
        ext = Mid$(fullPath, p)
    Else
        base = fullPath
        ext = ""
    End If

    candidate = fullPath
    i = 2

    Do While Len(Dir(candidate)) > 0

        candidate = base & " (" & i & ")" & ext
        i = i + 1

    Loop

    MakeUniquePath = candidate

End Function

Private Function ShortFilePart(ByVal filePart As String, _
                              ByVal maxLen As Long) As String

   filePart = CleanFilePart(filePart)

   If Len(filePart) > maxLen Then
       filePart = Left$(filePart, maxLen)
   End If

   If Len(Trim$(filePart)) = 0 Then
       filePart = "Birim"
   End If

   ShortFilePart = filePart

End Function
