Option Explicit

Private Const KOKPIT_SILINECEK_UST_SATIR_SAYISI As Long = 2

Public Sub Run_ButceKullanimOzeti()

    ButceKullanimOzeti_Core

End Sub


Public Sub Run_ButceKullanimOzeti_FromAll(ByVal pTedarikDosyaYolu As String, _
                                           ByVal pYatirimButceDosyaYolu As String, _
                                           ByVal pProjeBilgiDosyaYolu As String, _
                                           Optional ByVal pCiktiAnaKlasor As String = "", _
                                           Optional ByVal pButceTalepleriDosyaYolu As String = "")

    ButceKullanimOzeti_Core pTedarikDosyaYolu, _
                            pYatirimButceDosyaYolu, _
                            pProjeBilgiDosyaYolu, _
                            pCiktiAnaKlasor, _
                            pButceTalepleriDosyaYolu

End Sub


Private Sub ButceKullanimOzeti_Core(Optional ByVal pTedarikDosyaYolu As String = "", _
                                    Optional ByVal pYatirimButceDosyaYolu As String = "", _
                                    Optional ByVal pProjeBilgiDosyaYolu As String = "", _
                                    Optional ByVal pCiktiAnaKlasor As String = "", _
                                    Optional ByVal pButceTalepleriDosyaYolu As String = "")


   Dim wbTedarik As Workbook
   Dim wbYatirim As Workbook
   Dim wbYeni As Workbook

   Dim wsTedarik As Worksheet
   Dim wsYatirim As Worksheet
   Dim wsVeriSeti As Worksheet
   Dim wsKullanimOzet As Worksheet

   Dim tedarikDosyaYolu As String
   Dim yatirimDosyaYolu As String
   Dim projeBilgiDosyaYolu As String
   Dim butceTalepleriDosyaYolu As String

   Dim tarihMetni As String
   Dim ciktiKlasoru As String
   Dim ciktiDosyaYolu As String

   Dim tedarikHeaderRow As Long
   Dim yeniHeaderRow As Long

   Dim colTedButceNo As Long
   Dim colTedStatu As Long
   Dim colTedTutarUSD As Long

   Dim colYeniButceNo As Long
   Dim colYeniButce As Long
   Dim colYeniKullanilan As Long
   Dim colYeniBakiye As Long

   Dim colHesapHarcama As Long
   Dim colHesapTalep As Long

   Dim sonSatirTed As Long
   Dim sonSatirYatirim As Long
   Dim sonSutunYatirim As Long
   Dim sonSatirYeni As Long
   Dim sonSutunYeni As Long

   Dim dictHarcama As Object
   Dim dictTalep As Object

   Dim i As Long
   Dim butceNo As String
   Dim statu As String
   Dim tutarUSD As Double

   Dim eskiEkran As Boolean
   Dim eskiUyari As Boolean

   On Error GoTo HataYonetimi

   eskiEkran = Application.ScreenUpdating
   eskiUyari = Application.DisplayAlerts

   Application.ScreenUpdating = False
   Application.DisplayAlerts = False

   tarihMetni = Format(Date, "dd.mm.yyyy")

   If Len(Trim(pTedarikDosyaYolu)) > 0 Then
       tedarikDosyaYolu = pTedarikDosyaYolu
   Else
       tedarikDosyaYolu = DosyaSec("Lütfen KOKPİT sisteminden alınan Tedarik Kalemleri dosyasını seçiniz.")
   End If

   If Len(Trim(tedarikDosyaYolu)) = 0 Then GoTo TemizCikis

   If Len(Trim(pYatirimButceDosyaYolu)) > 0 Then
       yatirimDosyaYolu = pYatirimButceDosyaYolu
   Else
       yatirimDosyaYolu = DosyaSec("Lütfen Yatırım Bütçe dosyasını seçiniz.")
   End If
   
    If Len(Trim(pProjeBilgiDosyaYolu)) > 0 Then
    projeBilgiDosyaYolu = pProjeBilgiDosyaYolu
    Else
        projeBilgiDosyaYolu = DosyaSec("Lütfen Proje Kodu ve Proje Adı bilgilerini içeren kaynak dosyayı seçiniz.")
    End If
    
    If Len(Trim(projeBilgiDosyaYolu)) = 0 Then GoTo TemizCikis

   If Len(Trim(pButceTalepleriDosyaYolu)) > 0 Then
       butceTalepleriDosyaYolu = pButceTalepleriDosyaYolu
   Else
       butceTalepleriDosyaYolu = DosyaSec("Lütfen  Bütçe Talepleri dosyasını seçiniz.")
   End If

   If Len(Trim(butceTalepleriDosyaYolu)) = 0 Then GoTo TemizCikis


   If Len(Trim(yatirimDosyaYolu)) = 0 Then GoTo TemizCikis

   If Len(Trim(pCiktiAnaKlasor)) > 0 Then
       ciktiKlasoru = EnsureBudgetUsageFolder(pCiktiAnaKlasor, tarihMetni)
   Else
       ciktiKlasoru = EnsureBudgetUsageFolder(GetParentFolder(tedarikDosyaYolu), tarihMetni)
   End If

   Set wbTedarik = Workbooks.Open(tedarikDosyaYolu, ReadOnly:=True)
   Set wbYatirim = Workbooks.Open(yatirimDosyaYolu, ReadOnly:=True)

   Set wsTedarik = wbTedarik.Worksheets(1)
   Set wsYatirim = wbYatirim.Worksheets(1)

   ' ------------------------------------------------------------
   ' Tedarik dosyasından eski T/U algoritması için sözlükleri hazırla
   ' ------------------------------------------------------------
   tedarikHeaderRow = FindHeaderRow(wsTedarik, Array("Bütçe No", "Butce No"))
   If tedarikHeaderRow = 0 Then Err.Raise vbObjectError + 1001, , "Tedarik Kalemleri dosyasında 'Bütçe No' başlığı bulunamadı."

   colTedButceNo = FindHeaderCol(wsTedarik, tedarikHeaderRow, Array("Bütçe No", "Butce No", "Bütçe Numarası", "Butce Numarasi"))
   colTedStatu = FindHeaderCol(wsTedarik, tedarikHeaderRow, Array("Statü", "Statu", "Durum", "Talep Statüsü", "Talep Statusu", "Tedarik Statüsü", "Tedarik Statusu"))
   colTedTutarUSD = FindHeaderCol(wsTedarik, tedarikHeaderRow, Array( _
                                   "Talep Toplam USD", _
                                   "Toplam Talep USD", _
                                   "Toplam Talep Tutarı USD", _
                                   "Toplam Talep Tutari USD", _
                                   "Talep Tutarı USD", _
                                   "Talep Tutari USD", _
                                   "ABD $", _
                                   "USD"))

   If colTedButceNo = 0 Then Err.Raise vbObjectError + 1002, , "Tedarik Kalemleri dosyasında 'Bütçe No' kolonu bulunamadı."
   If colTedStatu = 0 Then Err.Raise vbObjectError + 1003, , "Tedarik Kalemleri dosyasında 'Statü' kolonu bulunamadı."
   If colTedTutarUSD = 0 Then Err.Raise vbObjectError + 1004, , "Tedarik Kalemleri dosyasında 'Talep Toplam USD' kolonu bulunamadı."

   Set dictHarcama = CreateObject("Scripting.Dictionary")
   Set dictTalep = CreateObject("Scripting.Dictionary")

   sonSatirTed = LastUsedRow(wsTedarik)

   For i = tedarikHeaderRow + 1 To sonSatirTed

       butceNo = NormalizeButceNo(wsTedarik.Cells(i, colTedButceNo).Value)
       statu = CStr(wsTedarik.Cells(i, colTedStatu).Value)
       tutarUSD = ToDoubleSafe(wsTedarik.Cells(i, colTedTutarUSD).Value)

       If Len(butceNo) > 0 Then

           If IsHarcamaStatus(statu) Then
               AddToDict dictHarcama, butceNo, tutarUSD
               AddToDict dictTalep, butceNo, tutarUSD

           ElseIf IsSadeceTalepStatus(statu) Then
               AddToDict dictTalep, butceNo, tutarUSD
           End If

       End If

   Next i

   ' ------------------------------------------------------------
   ' 01 - Yatırım Bütçe Kullanım Durum Özeti dosyası
   ' ------------------------------------------------------------
   ciktiDosyaYolu = ciktiKlasoru & Application.PathSeparator & _
                    "01 - Yatırım Bütçe Kullanım Durum Özeti - " & tarihMetni & ".xlsx"

   Set wbYeni = Workbooks.Add(xlWBATWorksheet)

   Set wsVeriSeti = wbYeni.Worksheets(1)
   wsVeriSeti.Name = "01 - Veri Seti"

   Set wsKullanimOzet = wbYeni.Worksheets.Add(After:=wsVeriSeti)
   wsKullanimOzet.Name = "02 - Kullanım Özeti"

   PrepareKokpitVeriSeti wsTedarik, wsVeriSeti, "YYSHUMA"

   sonSatirYatirim = LastUsedRow(wsYatirim)
   sonSutunYatirim = LastUsedCol(wsYatirim)

   wsYatirim.Range(wsYatirim.Cells(1, 1), wsYatirim.Cells(sonSatirYatirim, sonSutunYatirim)).Copy
   wsKullanimOzet.Range("A1").PasteSpecial xlPasteAll
   Application.CutCopyMode = False

   yeniHeaderRow = FindHeaderRow(wsKullanimOzet, Array("Bütçe No", "Butce No"))
   If yeniHeaderRow = 0 Then Err.Raise vbObjectError + 1005, , "Yatırım Bütçe dosyasında 'Bütçe No' başlığı bulunamadı."

   If yeniHeaderRow = 1 Then
       wsKullanimOzet.rows(1).Insert Shift:=xlDown
       yeniHeaderRow = yeniHeaderRow + 1
   End If

   DeleteColumnsByExactHeaders wsKullanimOzet, yeniHeaderRow, Array( _
       "Bütçe (TL)", _
       "Kullanılan(TL)", _
       "Kullanılan (TL)", _
       "Bakiye(TL)", _
       "Bakiye (TL)", _
       "%(TL)", _
       "% (TL)", _
       "Nakit TL", _
       "Üst Bütçe", _
       "Ust Butce")

   InsertMudurlukColumn wsKullanimOzet, yeniHeaderRow

   colYeniButceNo = FindHeaderCol(wsKullanimOzet, yeniHeaderRow, Array("Bütçe No", "Butce No", "Bütçe Numarası", "Butce Numarasi"))
   colYeniButce = FindHeaderCol(wsKullanimOzet, yeniHeaderRow, Array("Bütçe ($)", "Butce ($)", "Bütçe$", "Butce$", "Bütçe", "Butce"))
   colYeniKullanilan = FindHeaderCol(wsKullanimOzet, yeniHeaderRow, Array("Kullanılan($)", "Kullanilan($)", "Kullanılan ($)", "Kullanilan ($)", "Kullanılan", "Kullanilan"))
   colYeniBakiye = FindHeaderCol(wsKullanimOzet, yeniHeaderRow, Array("Bakiye($)", "Bakiye ($)", "Bakiye"))

   If colYeniButceNo = 0 Then Err.Raise vbObjectError + 1006, , "Yatırım Bütçe dosyasında 'Bütçe No' kolonu bulunamadı."
   If colYeniButce = 0 Then Err.Raise vbObjectError + 1007, , "Yatırım Bütçe dosyasında 'Bütçe ($)' kolonu bulunamadı."
   If colYeniKullanilan = 0 Then Err.Raise vbObjectError + 1008, , "Yatırım Bütçe dosyasında 'Kullanılan($)' kolonu bulunamadı."
   If colYeniBakiye = 0 Then Err.Raise vbObjectError + 1009, , "Yatırım Bütçe dosyasında 'Bakiye($)' kolonu bulunamadı."

   sonSutunYeni = LastUsedCol(wsKullanimOzet)

   colHesapHarcama = sonSutunYeni + 1
   colHesapTalep = sonSutunYeni + 2

   wsKullanimOzet.Cells(yeniHeaderRow, colHesapHarcama).Value = "Hesaplanan Harcama Tutarı USD"
   wsKullanimOzet.Cells(yeniHeaderRow, colHesapTalep).Value = "Hesaplanan Talep Toplam Tutarı USD"
   wsKullanimOzet.Cells(yeniHeaderRow, colHesapHarcama).Font.Bold = True
   wsKullanimOzet.Cells(yeniHeaderRow, colHesapTalep).Font.Bold = True

   sonSatirYeni = LastUsedRow(wsKullanimOzet)

   For i = yeniHeaderRow + 1 To sonSatirYeni

       butceNo = NormalizeButceNo(wsKullanimOzet.Cells(i, colYeniButceNo).Value)

       If Len(butceNo) > 0 Then

           If dictHarcama.Exists(butceNo) Then
               wsKullanimOzet.Cells(i, colHesapHarcama).Value = dictHarcama(butceNo)
           Else
               wsKullanimOzet.Cells(i, colHesapHarcama).Value = 0
           End If

           If dictTalep.Exists(butceNo) Then
               wsKullanimOzet.Cells(i, colHesapTalep).Value = dictTalep(butceNo)
           Else
               wsKullanimOzet.Cells(i, colHesapTalep).Value = 0
           End If

       End If

   Next i

   wsKullanimOzet.Cells(yeniHeaderRow - 1, colYeniButce).Formula = "=SUM(" & _
       wsKullanimOzet.Range(wsKullanimOzet.Cells(yeniHeaderRow + 1, colYeniButce), wsKullanimOzet.Cells(sonSatirYeni, colYeniButce)).Address(False, False) & ")"

   wsKullanimOzet.Cells(yeniHeaderRow - 1, colYeniKullanilan).Formula = "=SUM(" & _
       wsKullanimOzet.Range(wsKullanimOzet.Cells(yeniHeaderRow + 1, colYeniKullanilan), wsKullanimOzet.Cells(sonSatirYeni, colYeniKullanilan)).Address(False, False) & ")"

   wsKullanimOzet.Cells(yeniHeaderRow - 1, colYeniBakiye).Formula = "=SUM(" & _
       wsKullanimOzet.Range(wsKullanimOzet.Cells(yeniHeaderRow + 1, colYeniBakiye), wsKullanimOzet.Cells(sonSatirYeni, colYeniBakiye)).Address(False, False) & ")"

   wsKullanimOzet.Cells(yeniHeaderRow - 1, colHesapHarcama).Formula = "=SUM(" & _
       wsKullanimOzet.Range(wsKullanimOzet.Cells(yeniHeaderRow + 1, colHesapHarcama), wsKullanimOzet.Cells(sonSatirYeni, colHesapHarcama)).Address(False, False) & ")"

   wsKullanimOzet.Cells(yeniHeaderRow - 1, colHesapTalep).Formula = "=SUM(" & _
       wsKullanimOzet.Range(wsKullanimOzet.Cells(yeniHeaderRow + 1, colHesapTalep), wsKullanimOzet.Cells(sonSatirYeni, colHesapTalep)).Address(False, False) & ")"

   wsKullanimOzet.Range(wsKullanimOzet.Cells(yeniHeaderRow - 1, colYeniButce), _
                        wsKullanimOzet.Cells(yeniHeaderRow - 1, colHesapTalep)).Font.Bold = True

   ApplyNumberFormat wsKullanimOzet, yeniHeaderRow, sonSatirYeni, _
       Array(colYeniButce, colYeniKullanilan, colYeniBakiye, colHesapHarcama, colHesapTalep)

   wsKullanimOzet.Range(wsKullanimOzet.Cells(yeniHeaderRow, 1), wsKullanimOzet.Cells(yeniHeaderRow, colHesapTalep)).Font.Bold = True
   wsKullanimOzet.Range(wsKullanimOzet.Cells(yeniHeaderRow, 1), wsKullanimOzet.Cells(sonSatirYeni, colHesapTalep)).Borders.LineStyle = xlContinuous

   If wsKullanimOzet.AutoFilterMode Then wsKullanimOzet.AutoFilterMode = False
   wsKullanimOzet.Range(wsKullanimOzet.Cells(yeniHeaderRow, 1), wsKullanimOzet.Cells(sonSatirYeni, colHesapTalep)).AutoFilter

   wsKullanimOzet.Columns.AutoFit

    AutoFitAllSheets wbYeni
   wbYeni.SaveAs Filename:=ciktiDosyaYolu, FileFormat:=xlOpenXMLWorkbook
   wbYeni.Close SaveChanges:=False
   Set wbYeni = Nothing

   ' ------------------------------------------------------------
   ' 02 - İşletme Bütçe Kullanım Durum Özeti dosyası
   ' ------------------------------------------------------------
   CreateIsletmeButceKullanimDosyasi wsTedarik, ciktiKlasoru, tarihMetni, projeBilgiDosyaYolu, butceTalepleriDosyaYolu

TemizCikis:
   On Error Resume Next

   If Not wbYeni Is Nothing Then wbYeni.Close SaveChanges:=False
   If Not wbTedarik Is Nothing Then wbTedarik.Close SaveChanges:=False
   If Not wbYatirim Is Nothing Then wbYatirim.Close SaveChanges:=False

   Application.DisplayAlerts = eskiUyari
   Application.ScreenUpdating = eskiEkran

   If Len(Trim(ciktiKlasoru)) > 0 Then
       MsgBox "Bütçe Kullanım Durumu dosyaları başarıyla oluşturuldu." & vbCrLf & vbCrLf & _
              "Kayıt klasörü:" & vbCrLf & ciktiKlasoru, vbInformation, "İşlem Tamamlandı"
   End If

   Exit Sub

HataYonetimi:
   On Error Resume Next

   If Not wbYeni Is Nothing Then wbYeni.Close SaveChanges:=False
   If Not wbTedarik Is Nothing Then wbTedarik.Close SaveChanges:=False
   If Not wbYatirim Is Nothing Then wbYatirim.Close SaveChanges:=False

   Application.DisplayAlerts = eskiUyari
   Application.ScreenUpdating = eskiEkran

   MsgBox "İşlem sırasında hata oluştu:" & vbCrLf & vbCrLf & _
          Err.Description, vbCritical, "Bütçe Kullanım Özeti"

End Sub

Private Sub CreateIsletmeButceKullanimDosyasi(ByVal wsTedarik As Worksheet, _
                                               ByVal ciktiKlasoru As String, _
                                               ByVal tarihMetni As String, _
                                               ByVal projeBilgiDosyaYolu As String, _
                                               ByVal butceTalepleriDosyaYolu As String)


   Dim wbIsletme As Workbook
   Dim wsVeriSeti As Worksheet
   Dim wsGerceklesen As Worksheet
   Dim wsTalep As Worksheet
   Dim wsProjeGerceklesen As Worksheet
   Dim wsProjeTalep As Worksheet
   Dim wsGrupGerceklesen As Worksheet
   Dim wsGrupTalep As Worksheet
   Dim ciktiDosyaYolu As String
   Dim wbProjeBilgi As Workbook
   Dim wsProjeBilgi As Worksheet
   Dim wbButceTalepleri As Workbook
   Dim wsButceTalepleri As Worksheet
   Dim dictProjeAdi As Object
   Dim dictProjeKodu As Object


   ciktiDosyaYolu = ciktiKlasoru & Application.PathSeparator & _
                    "02 - İşletme Bütçe Kullanım Durum Özeti - " & tarihMetni & ".xlsx"
    Set wbProjeBilgi = Workbooks.Open(projeBilgiDosyaYolu, ReadOnly:=True)
    Set wsProjeBilgi = wbProjeBilgi.Worksheets(1)
    Set dictProjeAdi = BuildProjeAdiDict(wsProjeBilgi)
    Set dictProjeKodu = BuildProjeKoduDict(wsProjeBilgi)
    Set wbButceTalepleri = Workbooks.Open(butceTalepleriDosyaYolu, ReadOnly:=True)
    Set wsButceTalepleri = wbButceTalepleri.Worksheets(1)
    
                

   Set wbIsletme = Workbooks.Add(xlWBATWorksheet)

   Set wsVeriSeti = wbIsletme.Worksheets(1)
   wsVeriSeti.Name = "01 - Veri Seti"

   Set wsGerceklesen = wbIsletme.Worksheets.Add(After:=wsVeriSeti)
   wsGerceklesen.Name = "02 - Kullanım Özeti-GERÇEKLEŞEN"

   Set wsTalep = wbIsletme.Worksheets.Add(After:=wsGerceklesen)
   wsTalep.Name = "03 - Kullanım Özeti-TALEP"

   Set wsProjeGerceklesen = wbIsletme.Worksheets.Add(After:=wsTalep)
   wsProjeGerceklesen.Name = "04- Proje Kullanım - GERÇEKLEŞEN"

   Set wsProjeTalep = wbIsletme.Worksheets.Add(After:=wsProjeGerceklesen)
   wsProjeTalep.Name = "05- Proje Kullanım - TALEP"

   Set wsGrupGerceklesen = wbIsletme.Worksheets.Add(After:=wsProjeTalep)
   wsGrupGerceklesen.Name = "06- Grup Tasnifli Bütçe GERÇEKLEŞEN"

   Set wsGrupTalep = wbIsletme.Worksheets.Add(After:=wsGrupGerceklesen)
   wsGrupTalep.Name = "07 - Grup Tasnifli Bütçe TALEP"

   PrepareKokpitVeriSeti wsTedarik, wsVeriSeti, "IPTSHUMA"

    CreateIsletmeKullanimOzeti wsVeriSeti, wsGerceklesen, False, "Gerçekleşen", dictProjeAdi
    CreateIsletmeKullanimOzeti wsVeriSeti, wsTalep, True, "Talep", dictProjeAdi
    CreateIsletmeKullanimOzeti wsVeriSeti, wsProjeGerceklesen, False, "Gerçekleşen", dictProjeAdi, True
    CreateIsletmeKullanimOzeti wsVeriSeti, wsProjeTalep, True, "Talep", dictProjeAdi, True
    CreateGrupTasnifliButceOzeti wsVeriSeti, wsButceTalepleri, wsGrupGerceklesen, False, "Gerçekleşen", dictProjeAdi
    CreateGrupTasnifliButceOzeti wsVeriSeti, wsButceTalepleri, wsGrupTalep, True, "Talep", dictProjeAdi


   AutoFitAllSheets wbIsletme
   wbIsletme.SaveAs Filename:=ciktiDosyaYolu, FileFormat:=xlOpenXMLWorkbook
   wbIsletme.Close SaveChanges:=False
   If Not wbProjeBilgi Is Nothing Then wbProjeBilgi.Close SaveChanges:=False
   If Not wbButceTalepleri Is Nothing Then wbButceTalepleri.Close SaveChanges:=False

End Sub


Private Sub CreateIsletmeKullanimOzeti(ByVal wsVeriSeti As Worksheet, _
                                      ByVal wsOzet As Worksheet, _
                                      ByVal humaStatuleriDahil As Boolean, _
                                      ByVal baslikTipi As String, _
                                      ByVal dictProjeAdi As Object, _
                                      Optional ByVal projeBazliOzet As Boolean = False)

   Dim headerRow As Long
   Dim lastRow As Long
   Dim i As Long

   Dim colTalepEdenBolum As Long
   Dim colProje As Long
   Dim colStatu As Long
   Dim colGirisTarihi As Long
   Dim colTalepToplamUSD As Long

   Dim dict As Object
   Dim key As String

   Dim talepEdenBolum As String
   Dim temizBirim As String
   Dim baskanlik As String
   Dim mudurluk As String
   Dim birim As String
   Dim proje As String
   Dim projeAdi As String
   Dim statu As String
   Dim ayNo As Integer
   Dim tutar As Double
   Dim toplam As Double

   Dim arr As Variant
   Dim satir As Long
   Dim k As Variant
   Dim m As Long
   Dim lastOzetRow As Long

   Set dict = CreateObject("Scripting.Dictionary")

   headerRow = 1
   lastRow = LastUsedRow(wsVeriSeti)

   colTalepEdenBolum = FindHeaderCol(wsVeriSeti, headerRow, Array("Talep Eden Bölüm", "Talep Eden Bolum"))
   colProje = FindHeaderCol(wsVeriSeti, headerRow, Array("Proje"))
   colStatu = FindHeaderCol(wsVeriSeti, headerRow, Array("Statü", "Statu", "Durum", "Talep Statüsü", "Talep Statusu", "Tedarik Statüsü", "Tedarik Statusu"))
   colGirisTarihi = FindHeaderCol(wsVeriSeti, headerRow, Array("Giriş Tarihi", "Giris Tarihi"))
   colTalepToplamUSD = FindHeaderCol(wsVeriSeti, headerRow, Array("Talep Toplam USD"))

   If colTalepEdenBolum = 0 Then Err.Raise vbObjectError + 1101, , "'Talep Eden Bölüm' sütunu bulunamadı."
   If colProje = 0 Then Err.Raise vbObjectError + 1102, , "'Proje' sütunu bulunamadı."
   If colStatu = 0 Then Err.Raise vbObjectError + 1103, , "'Statü' sütunu bulunamadı."
   If colGirisTarihi = 0 Then Err.Raise vbObjectError + 1104, , "'Giriş Tarihi' sütunu bulunamadı."
   If colTalepToplamUSD = 0 Then Err.Raise vbObjectError + 1105, , "'Talep Toplam USD' sütunu bulunamadı."

   For i = headerRow + 1 To lastRow

       statu = CStr(wsVeriSeti.Cells(i, colStatu).Value)

       If IsIsletmeOzetStatus(statu, humaStatuleriDahil) Then

           ayNo = GetMonthNoFromValue(wsVeriSeti.Cells(i, colGirisTarihi).Value)

           If ayNo >= 1 And ayNo <= 12 Then

               talepEdenBolum = CStr(wsVeriSeti.Cells(i, colTalepEdenBolum).Value)
               temizBirim = CleanTalepEdenBolum(talepEdenBolum)

               If projeBazliOzet Then
                   baskanlik = "V"
                   mudurluk = "KONSOLİDE"
                   birim = "KONSOLİDE"
               Else
                   If Len(temizBirim) >= 1 Then
                       baskanlik = Left(temizBirim, 1)
                   Else
                       baskanlik = ""
                   End If

                   If Len(temizBirim) >= 3 Then
                       mudurluk = Left(temizBirim, 3)
                   Else
                       mudurluk = temizBirim
                   End If

                   birim = temizBirim
               End If

               proje = Trim(CStr(wsVeriSeti.Cells(i, colProje).Value))
               tutar = ToDoubleSafe(wsVeriSeti.Cells(i, colTalepToplamUSD).Value)

               If Not dictProjeAdi Is Nothing Then
                   If dictProjeAdi.Exists(NormalizeProjectCode(proje)) Then
                       projeAdi = dictProjeAdi(NormalizeProjectCode(proje))
                   Else
                       projeAdi = ""
                   End If
               Else
                   projeAdi = ""
               End If

               If projeBazliOzet Then
                   ' Proje bazlı özetlerde aynı proje tek satırdır.
                   key = NormalizeProjectCode(proje)
               Else

Private Sub DeleteColumnsByExactHeaders(ByVal ws As Worksheet, ByVal headerRow As Long, ByVal headersToDelete As Variant)

   Dim lastCol As Long
   Dim c As Long
   Dim h As Variant
   Dim cellText As String
   Dim targetText As String
   Dim shouldDelete As Boolean

   lastCol = ws.Cells(headerRow, ws.Columns.Count).End(xlToLeft).Column

   For c = lastCol To 1 Step -1

       cellText = NormalizeText(ws.Cells(headerRow, c).Value)
       shouldDelete = False

       For Each h In headersToDelete
           targetText = NormalizeText(CStr(h))

           If cellText = targetText Then
               shouldDelete = True
               Exit For
           End If
       Next h

       If shouldDelete Then
           ws.Columns(c).Delete
       End If

   Next c

End Sub

Private Sub InsertMudurlukColumn(ByVal ws As Worksheet, ByVal headerRow As Long)

   Dim colBolum As Long
   Dim colMudurluk As Long
   Dim lastRow As Long
   Dim i As Long
   Dim bolumDegeri As String

   colBolum = FindHeaderCol(ws, headerRow, Array("Bölüm", "Bolum"))

   If colBolum = 0 Then
       Err.Raise vbObjectError + 1020, , "Yatırım Bütçe dosyasında 'Bölüm' başlığı bulunamadı. Müdürlük sütunu oluşturulamadı."
   End If

   ws.Columns(colBolum).Insert Shift:=xlToRight

   colMudurluk = colBolum
   colBolum = colBolum + 1

   ws.Cells(headerRow, colMudurluk).Value = "Müdürlük"
   ws.Cells(headerRow, colMudurluk).Font.Bold = True

   lastRow = LastUsedRow(ws)

   For i = headerRow + 1 To lastRow

       bolumDegeri = Trim(CStr(ws.Cells(i, colBolum).Value))

       If Len(bolumDegeri) >= 3 Then
           ws.Cells(i, colMudurluk).Value = Left(bolumDegeri, 3)
       Else
           ws.Cells(i, colMudurluk).Value = bolumDegeri
       End If

   Next i

End Sub

Private Function IsHarcamaStatus(ByVal statu As String) As Boolean

   Dim s As String
   s = NormalizeText(statu)

   If InStr(1, s, "PR KESILDI", vbTextCompare) > 0 And _
      InStr(1, s, "PO HENUZ KESILMEDI", vbTextCompare) > 0 Then
       IsHarcamaStatus = True
       Exit Function
   End If

   If InStr(1, s, "TESLIM EDILDI", vbTextCompare) > 0 Then

Private Function GetMonthNoFromValue(ByVal valueIn As Variant) As Integer

   Dim s As String
   Dim parts As Variant

   On Error GoTo Hata

   If IsDate(valueIn) Then
       GetMonthNoFromValue = Month(CDate(valueIn))
       Exit Function
   End If

   s = Trim(CStr(valueIn))

   If Len(s) = 0 Then
       GetMonthNoFromValue = 0
       Exit Function
   End If

   If InStr(1, s, ".", vbTextCompare) > 0 Then

       parts = Split(s, ".")

       If UBound(parts) >= 2 Then
           GetMonthNoFromValue = CInt(parts(1))
           Exit Function
       End If

   End If

   If InStr(1, s, "-", vbTextCompare) > 0 Then

       parts = Split(s, "-")

       If UBound(parts) >= 2 Then
           GetMonthNoFromValue = CInt(parts(1))
           Exit Function
       End If

   End If

Hata:
   GetMonthNoFromValue = 0

End Function

Private Function EnsureBudgetUsageFolder(ByVal baseFolder As String, ByVal tarihMetni As String) As String

   Dim folderName As String
   Dim normalizedBaseName As String
   Dim baseName As String
   Dim p As Long

   folderName = "Bütçe Kullanım Durumu " & tarihMetni
   baseFolder = Trim(baseFolder)

   If Right(baseFolder, 1) = Application.PathSeparator Then
       baseFolder = Left(baseFolder, Len(baseFolder) - 1)
   End If

   p = InStrRev(baseFolder, Application.PathSeparator)

   If p > 0 Then
       baseName = Mid(baseFolder, p + 1)
   Else
       baseName = baseFolder
   End If

   normalizedBaseName = NormalizeText(baseName)

   If normalizedBaseName = NormalizeText(folderName) Then
       EnsureBudgetUsageFolder = baseFolder
   Else
       EnsureBudgetUsageFolder = baseFolder & Application.PathSeparator & folderName
   End If

   If Dir(EnsureBudgetUsageFolder, vbDirectory) = "" Then
       MkDir EnsureBudgetUsageFolder
   End If

End Function

Private Function FindHeaderRow(ByVal ws As Worksheet, ByVal possibleHeaders As Variant) As Long

   Dim r As Long
   Dim c As Long
   Dim lastCol As Long
   Dim h As Variant
   Dim cellText As String

   For r = 1 To 15

       lastCol = ws.Cells(r, ws.Columns.Count).End(xlToLeft).Column

       For c = 1 To lastCol

           cellText = NormalizeText(ws.Cells(r, c).Value)

           For Each h In possibleHeaders
               If cellText = NormalizeText(CStr(h)) Then
                   FindHeaderRow = r
                   Exit Function
               End If
           Next h

       Next c

   Next r

   FindHeaderRow = 0

End Function

Private Function FindHeaderCol(ByVal ws As Worksheet, ByVal headerRow As Long, ByVal possibleHeaders As Variant) As Long

   Dim c As Long
   Dim lastCol As Long
   Dim h As Variant
   Dim cellText As String
   Dim targetText As String

   lastCol = ws.Cells(headerRow, ws.Columns.Count).End(xlToLeft).Column

   For c = 1 To lastCol

       cellText = NormalizeText(ws.Cells(headerRow, c).Value)

       For Each h In possibleHeaders
           targetText = NormalizeText(CStr(h))

           If cellText = targetText Then
               FindHeaderCol = c
               Exit Function
           End If
       Next h

   Next c

   For c = 1 To lastCol

       cellText = NormalizeText(ws.Cells(headerRow, c).Value)

       For Each h In possibleHeaders
           targetText = NormalizeText(CStr(h))

           If Len(targetText) > 2 Then
               If InStr(1, cellText, targetText, vbTextCompare) > 0 Then
                   FindHeaderCol = c
                   Exit Function
               End If
           End If
       Next h

   Next c

   FindHeaderCol = 0

End Function

Private Function LastUsedRow(ByVal ws As Worksheet) As Long

   Dim f As Range

   Set f = ws.Cells.Find(What:="*", _
                         After:=ws.Range("A1"), _
                         LookIn:=xlFormulas, _
                         LookAt:=xlPart, _
                         SearchOrder:=xlByRows, _
                         SearchDirection:=xlPrevious, _
                         MatchCase:=False)

   If f Is Nothing Then
       LastUsedRow = 1
   Else
       LastUsedRow = f.Row
   End If

End Function

Private Function LastUsedCol(ByVal ws As Worksheet) As Long

   Set f = ws.Cells.Find(What:="*", _
                         After:=ws.Range("A1"), _
                         LookIn:=xlFormulas, _
                         LookAt:=xlPart, _
                         SearchOrder:=xlByColumns, _
                         SearchDirection:=xlPrevious, _
                         MatchCase:=False)

   If f Is Nothing Then
       LastUsedCol = 1
   Else
       LastUsedCol = f.Column
   End If

End Function

Private Function NormalizeText(ByVal valueIn As Variant) As String

   Dim s As String

   If IsError(valueIn) Then
       NormalizeText = ""
       Exit Function
   End If

   s = CStr(valueIn)

   s = Replace(s, Chr(10), " ")
   s = Replace(s, Chr(13), " ")
   s = Replace(s, vbTab, " ")

   s = Trim(s)

   Do While InStr(s, "  ") > 0
       s = Replace(s, "  ", " ")
   Loop

   s = UCase$(s)

   s = Replace(s, "İ", "I")
   s = Replace(s, "İ", "I")
   s = Replace(s, "Ş", "S")
   s = Replace(s, "Ğ", "G")
   s = Replace(s, "Ü", "U")
   s = Replace(s, "Ö", "O")
   s = Replace(s, "Ç", "C")
   s = Replace(s, "ı", "I")
   s = Replace(s, "ş", "S")
   s = Replace(s, "ğ", "G")
   s = Replace(s, "ü", "U")
   s = Replace(s, "ö", "O")
   s = Replace(s, "ç", "C")

   NormalizeText = s

End Function

Private Function ToDoubleSafe(ByVal valueIn As Variant) As Double

   Dim s As String
   Dim lastComma As Long
   Dim lastDot As Long
   Dim negativeValue As Boolean

   If IsError(valueIn) Then
       ToDoubleSafe = 0
       Exit Function
   End If

   If IsNumeric(valueIn) Then
       ToDoubleSafe = CDbl(valueIn)
       Exit Function
   End If

   s = Trim(CStr(valueIn))

   If Len(s) = 0 Then
       ToDoubleSafe = 0
       Exit Function
   End If

   If Left(s, 1) = "(" And Right(s, 1) = ")" Then
       negativeValue = True
       s = Mid(s, 2, Len(s) - 2)
   End If

   s = Replace(s, "USD", "", , , vbTextCompare)
   s = Replace(s, "ABD", "", , , vbTextCompare)
   s = Replace(s, "$", "")
   s = Replace(s, " ", "")

   lastComma = InStrRev(s, ",")
   lastDot = InStrRev(s, ".")

   If lastComma > 0 And lastDot > 0 Then

       If lastComma > lastDot Then
           s = Replace(s, ".", "")
           s = Replace(s, ",", ".")
       Else
           s = Replace(s, ",", "")
       End If

   ElseIf lastComma > 0 Then

       s = Replace(s, ".", "")
       s = Replace(s, ",", ".")

   Else

       s = Replace(s, ",", "")

   End If

   If IsNumeric(s) Then
       ToDoubleSafe = CDbl(s)
       If negativeValue Then ToDoubleSafe = -ToDoubleSafe
   Else
       ToDoubleSafe = 0
   End If

End Function

Private Function BuildProjeAdiDict(ByVal wsProjeBilgi As Worksheet) As Object

    Dim dict As Object
    Dim lastRow As Long
    Dim headerRow As Long
    Dim colProjeKodu As Long
    Dim colProjeAdi As Long

    Set dict = CreateObject("Scripting.Dictionary")

    lastRow = LastUsedRow(wsProjeBilgi)

    headerRow = FindHeaderRow(wsProjeBilgi, Array("Proje Kodu", "Proje Kod", "Proje"))

    If headerRow > 0 Then
        colProjeKodu = FindHeaderCol(wsProjeBilgi, headerRow, Array("Proje Kodu", "Proje Kod", "Proje"))
        colProjeAdi = FindHeaderCol(wsProjeBilgi, headerRow, Array("Proje Adı", "Proje Adi"))

        If colProjeKodu > 0 And colProjeAdi > 0 Then
            AddProjeAdiRangeToDict dict, wsProjeBilgi, headerRow + 1, lastRow, colProjeKodu, colProjeAdi
        End If
    End If

    ' Eski/standart kaynak dosya yerleşimleri için fallback:
    ' A sütunu = Proje Kodu, G sütunu = Proje Adı
    ' B sütunu = Proje Kodu, D sütunu = Proje Adı
    AddProjeAdiRangeToDict dict, wsProjeBilgi, 2, lastRow, 1, 7
    AddProjeAdiRangeToDict dict, wsProjeBilgi, 2, lastRow, 2, 4

    Set BuildProjeAdiDict = dict

End Function

Private Sub AddProjeAdiRangeToDict(ByVal dict As Object, _
                                    ByVal wsProjeBilgi As Worksheet, _
                                    ByVal firstRow As Long, _
                                    ByVal lastRow As Long, _
                                    ByVal colProjeKodu As Long, _
                                    ByVal colProjeAdi As Long)

    Dim i As Long
    Dim projeKodu As String
    Dim projeAdi As String

    If firstRow > lastRow Then Exit Sub
    If colProjeKodu <= 0 Or colProjeAdi <= 0 Then Exit Sub

    For i = firstRow To lastRow

        projeKodu = NormalizeProjectCode(wsProjeBilgi.Cells(i, colProjeKodu).Value)
        projeAdi = Trim(CStr(wsProjeBilgi.Cells(i, colProjeAdi).Value))

        If Len(projeKodu) > 0 Then
            If Not dict.Exists(projeKodu) Then
                dict.Add projeKodu, projeAdi
            ElseIf Len(Trim(CStr(dict(projeKodu)))) = 0 And Len(projeAdi) > 0 Then
                dict(projeKodu) = projeAdi
            End If
        End If

    Next i

End Sub

Private Sub KeepOnlyKaynakSistemInVeriSeti(ByVal ws As Worksheet, ByVal hedefKaynakSistem As String)

   Dim headerRow As Long
   Dim colKaynakSistem As Long
   Dim lastRow As Long
   Dim i As Long
   Dim deger As String
   Dim hedef As String

   headerRow = 1

   colKaynakSistem = FindHeaderCol(ws, headerRow, Array("Kaynak Sistem"))

   If colKaynakSistem = 0 Then
       Err.Raise vbObjectError + 1201, , "'Kaynak Sistem' sütunu bulunamadı. Veri Seti filtreleme işlemi uygulanamadı."
   End If

   hedef = NormalizeText(hedefKaynakSistem)
   lastRow = LastUsedRow(ws)

   For i = lastRow To headerRow + 1 Step -1

       deger = NormalizeText(ws.Cells(i, colKaynakSistem).Value)

       If deger <> hedef Then
           ws.rows(i).Delete
       End If

   Next i

End Sub
